module reg_file(input clk, input reg_w, input [4:0] a, input [4:0] b, input [4:0] w, input [31:0] data_in, output [31:0] rd1, output [31:0] rd2);

reg [31:0] reg_bank [31:0];
integer i;

initial begin
for (i=0;i<32;i=i+1)
	reg_bank[i] = 32'b0;
end



assign rd1 = (reg_w && w == a && a != 0) ? data_in : reg_bank[a];
assign rd2 = (reg_w && w == b && b != 0) ? data_in : reg_bank[b];


always @(posedge clk)
begin
if (reg_w && w)	reg_bank[w] <= data_in;
end

endmodule
