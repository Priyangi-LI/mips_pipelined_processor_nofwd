module hazard_detect_unit (input id_ex_mem_r, input [4:0] id_ex_rt,input [4:0] if_id_rs,input [4:0] if_id_rt, output reg pc_write, output reg if_id_write, output reg ctrl_flush);

always @(*) begin
if (id_ex_mem_r == 1 && (id_ex_rt == if_id_rs || id_ex_rt == if_id_rt) && id_ex_rt!= 0) begin

pc_write = 0; if_id_write = 0; ctrl_flush = 1;
end

else begin
pc_write = 1; if_id_write = 1; ctrl_flush = 0;
end

end
endmodule
