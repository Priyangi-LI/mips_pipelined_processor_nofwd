module if_id (input clk, input enable, input flush, input rst, input [31:0] pc_plus_4_in, input [31:0] instr_in, output reg [31:0] pc_plus_4_out, output reg [31:0] instr_out);

always @(posedge clk)
begin 

if (rst)
begin	instr_out <= 32'b0;	
pc_plus_4_out <= 32'b0;
end

else if (flush)
begin	instr_out <= 32'b0;	
pc_plus_4_out <= 32'b0;
end


else if (enable) begin
instr_out<=instr_in; pc_plus_4_out <= pc_plus_4_in;

end
end

endmodule
