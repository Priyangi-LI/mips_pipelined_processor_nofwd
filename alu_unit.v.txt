module alu(input [31:0] a, input [31:0] b, input [3:0] alu_ctrl,output reg [31:0] y, output zero);



always @(*)
begin


case(alu_ctrl)
4'b0000: y = a & b; //and
4'b0001: y = a | b; //or
4'b0010: y = a + b;  //add
4'b0110: y = a - b;  //sub
4'b0111: y = (a<b)? 32'b1: 32'b0; //slt

default : y = 32'b0;
endcase


end
assign zero = (y==0)? 1:0;
endmodule
