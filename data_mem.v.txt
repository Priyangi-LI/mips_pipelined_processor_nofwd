module data_mem(input clk, input mem_r, input mem_w, input [31:0] addr, input [31:0] data_in, output [31:0] data_w); 
reg [31:0] mem_bank [31:0]; 

integer i;
initial begin
for (i=0;i<31;i=i+1)	mem_bank[i] = 32'b0;
end

assign data_w = mem_bank[addr>>2]; 

always @(posedge clk) 
begin 

if (mem_w) mem_bank[addr>>2] <= data_in; 
 

end 
endmodule
